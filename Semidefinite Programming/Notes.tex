\section{Semidefinite Programming}
\begin{definition}
	A symmetric $n \times n$ matrix $A$ is PSD if $x^TAx \geq 0 \forall x$
\end{definition}
\begin{theorem}
The following are equivalent:
\begin{enumerate}
	\item $x^TAx \geq 0 \forall x$
	\item $A = \sum_{i = 0}^{n}\lambda_iv_iv_i^T$, where $\lambda_i \in \mathbb{R}^{+}$ and $v_i$ are orthonormal.
	\item $A = B^TB$ for some $B$
\end{enumerate}
\begin{proof}
	\begin{enumerate}
		\item $1) \implies 2)$: 
		\item $2) \implies 3)$:
		\item $3) \implies 1)$:
	\end{enumerate}
\end{proof}
\end{theorem}
